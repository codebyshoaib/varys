# Crontab template for the personal agent
# After running /setup, replace REPO_PATH with your actual repo path
# Run: crontab -e  then paste these lines

REPO_PATH=/path/to/your/personal-agent

# Slack poller — every 30 minutes
*/30 * * * * nice -n 15 $REPO_PATH/.claude/hooks/cron-wrap.sh poller python3 $REPO_PATH/.claude/hooks/slack-poller.py

# Daily tasks — 8am
0 8 * * * $REPO_PATH/.claude/hooks/cron-wrap.sh daily $REPO_PATH/varys-daily.sh

# Weekly report — Monday 9am
0 9 * * 1 $REPO_PATH/.claude/hooks/cron-wrap.sh weekly $REPO_PATH/varys-weekly-report.sh

# Slack listener — on reboot
@reboot python3 $REPO_PATH/.claude/hooks/varys-slack-listener.py >> /tmp/agent-slack-listener.log 2>&1

# Team Orchestrator — agent-free 270s heartbeat (replaces `/loop 270s`). Pure-Python
# tick; `claude` is spawned only when events are pending. Needs GITHUB_TOKEN/REPO in
# ~/.agent-config.json (cron has no session env).
@reboot cd $REPO_PATH && ./varys-orchestrator-loop.sh >> /tmp/varys-orchestrator-tick.log 2>&1

# Learning loop — 2am daily
0 2 * * * $REPO_PATH/.claude/hooks/cron-wrap.sh learn $REPO_PATH/varys-learn.sh

# Self-healer — every 10 minutes
*/10 * * * * nice -n 15 $REPO_PATH/.claude/hooks/cron-wrap.sh self-healer python3 $REPO_PATH/.claude/hooks/varys-self-healer.py

# Job finder — every 35 minutes
5,35 * * * * nice -n 15 $REPO_PATH/.claude/hooks/cron-wrap.sh job-finder python3 $REPO_PATH/.claude/hooks/job-finder.py

# Content scheduler — 6am daily
0 6 * * * $REPO_PATH/.claude/hooks/cron-wrap.sh content python3 $REPO_PATH/.claude/hooks/content-scheduler.py

# Notion map updater — 2:30am daily
30 2 * * * $REPO_PATH/.claude/hooks/cron-wrap.sh notion-map python3 $REPO_PATH/.claude/hooks/notion-map-updater.py --mode daily

# Observer — hourly at :15
15 * * * * $REPO_PATH/.claude/hooks/cron-wrap.sh observer python3 $REPO_PATH/.claude/hooks/varys-observer.py
